OOP Concepts

Classes and Objects


#include <iostream>
#include <string>

class Dog {
public:
    std::string name;
    int age;

    Dog(std::string n, int a) : name(n), age(a) {}

    void bark() {
        std::cout << "Woof!" << std::endl;
    }
};

int main() {
    Dog myDog("Buddy", 5);
    myDog.bark();
    std::cout << myDog.name << " is " << myDog.age << " years old." << std::endl;
    return 0;
}

Inheritance


#include <iostream>

class Animal {
public:
    void eat() {
        std::cout << "This animal eats food." << std::endl;
    }
};

class Dog : public Animal {
public:
    void bark() {
        std::cout << "Woof!" << std::endl;
    }
};

int main() {
    Dog myDog;
    myDog.eat();
    myDog.bark();
    return 0;
}

Polymorphism


#include <iostream>

class Animal {
public:
    virtual void sound() {
        std::cout << "This animal makes a sound." << std::endl;
    }
};

class Dog : public Animal {
public:
    void sound() override {
        std::cout << "Woof!" << std::endl;
    }
};

class Cat : public Animal {
public:
    void sound() override {
        std::cout << "Meow!" << std::endl;
    }
};

int main() {
    Animal* myDog = new Dog();
    Animal* myCat = new Cat();

    myDog->sound();
    myCat->sound();

    delete myDog;
    delete myCat;

    return 0;
}

Encapsulation


#include <iostream>

class Person {
private:
    std::string name;
    int age;

public:
    void setName(std::string n) {
        name = n;
    }

    std::string getName() {
        return name;
    }

    void setAge(int a) {
        age = a;
    }

    int getAge() {
        return age;
    }
};

int main() {
    Person person;
    person.setName("John");
    person.setAge(30);

    std::cout << person.getName() << " is " << person.getAge() << " years old." << std::endl;
    return 0;
}

Exceptions

Try-Catch Block


#include <iostream>
#include <exception>

int main() {
    try {
        throw std::runtime_error("An error occurred");
    } catch (const std::exception& e) {
        std::cout << "Exception: " << e.what() << std::endl;
    }
    return 0;
}

Custom Exceptions


#include <iostream>
#include <exception>

class CustomException : public std::exception {
private:
    std::string message;

public:
    CustomException(const std::string& msg) : message(msg) {}

    const char* what() const noexcept override {
        return message.c_str();
    }
};

int main() {
    try {
        throw CustomException("This is a custom exception");
    } catch (const CustomException& e) {
        std::cout << e.what() << std::endl;
    }
    return 0;
}

Templates

Function Templates

    return a + b;
}

    std::cout << "Int: " << add(2, 3) << std::endl;
    std::cout << "Double: " << add(2.5, 3.5) << std::endl;

}

Class Templates


#include <iostream>

template <typename T>
class Box {
private:
    T value;

public:
    Box(T v) : value(v) {}

    T getValue() {
        return value;
    }
};

int main() {
    Box<int> intBox(123);
    Box<std::string> strBox("Hello");

    std::cout << "Int value: " << intBox.getValue() << std::endl;
    std::cout << "String value: " << strBox.getValue() << std::endl;
    return 0;
}

File Handling

Reading from a File


#include <iostream>
#include <fstream>
#include <string>

int main() {
    std::ifstream infile("example.txt");
    std::string line;

    if (infile.is_open()) {
        while (std::getline(infile, line)) {
            std::cout << line << std::endl;
        }
        infile.close();
    } else {
        std::cout << "Unable to open file" << std::endl;
    }
    return 0;
}

Writing to a File


#include <iostream>
#include <fstream>

int main() {
    std::ofstream outfile("example.txt");

    if (outfile.is_open()) {
        outfile << "Hello, file!" << std::endl;
        outfile.close();
    } else {
        std::cout << "Unable to open file" << std::endl;
    }
    return 0;
}
