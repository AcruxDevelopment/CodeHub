Introducción

C++ es un lenguaje de programación potente y de alto rendimiento que soporta programación procedural, orientada a objetos y genérica. Se utiliza ampliamente para el desarrollo de sistemas/software, desarrollo de juegos y en aplicaciones críticas para el rendimiento.

Sintaxis de C++

Estructura Básica


#include <iostream>

int main() {
    std::cout << "¡Hola, Mundo!" << std::endl;
    return 0;
}

Conceptos OOP

Clases y Objetos


#include <iostream>
#include <string>

class Dog {
public:
    std::string name;
    int age;

    Dog(std::string n, int a) : name(n), age(a) {}

    void bark() {
        std::cout << "¡Guau!" << std::endl;
    }
};

int main() {
    Dog myDog("Buddy", 5);
    myDog.bark();
    std::cout << myDog.name << " tiene " << myDog.age << " años." << std::endl;
    return 0;
}

Herencia


#include <iostream>

class Animal {
public:
    void eat() {
        std::cout << "Este animal come comida." << std::endl;
    }
};

public:

        std::cout << "¡Guau!" << std::endl;
    }

int main() {
    Dog myDog;
    myDog.eat();
    myDog.bark();
    return 0;
}

Polimorfismo


#include <iostream>

class Animal {
public:
    virtual void sound() {
        std::cout << "Este animal hace un sonido." << std::endl;
    }
};

class Dog : public Animal {
public:
    void sound() override {
        std::cout << "¡Guau!" << std::endl;
    }
};

class Cat : public Animal {
public:
    void sound() override {
        std::cout << "¡Miau!" << std::endl;
    }
};

int main() {
    Animal* myDog = new Dog();
    Animal* myCat = new Cat();

    myDog->sound();
    myCat->sound();

    delete myDog;
    delete myCat;

    return 0;
}

Encapsulación


#include <iostream>

class Person {
private:
    std::string name;
    int age;

public:
    void setName(std::string n) {
        name = n;
    }

    std::string getName() {
        return name;
    }

    void setAge(int a) {
        age = a;
    }

    int getAge() {
        return age;
    }
};

int main() {
    Person person;
    person.setName("John");
    person.setAge(30);

    std::cout << person.getName() << " tiene " << person.getAge() << " años." << std::endl;
    return 0;
}

Excepciones

Bloque Try-Catch


#include <iostream>
#include <exception>

int main() {
    try {
        throw std::runtime_error("Ocurrió un error");
    } catch (const std::exception& e) {
        std::cout << "Excepción: " << e.what() << std::endl;
    }
    return 0;
}

Excepciones Personalizadas


#include <iostream>
#include <exception>

class CustomException : public std::exception {
private:
    std::string message;

public:
    CustomException(const std::string& msg) : message(msg) {}

    const char* what() const noexcept override {
        return message.c_str();
    }
};

int main() {
    try {
        throw CustomException("Esta es una excepción personalizada");
    } catch (const CustomException& e) {
        std::cout << e.what() << std::endl;
    }
    return 0;
}

Colecciones

Ejemplo de Vector


#include <iostream>
#include <vector>

int main() {
    std::vector<std::string> vec = {"Manzana", "Plátano", "Cereza"};

Templates

Templates de Función


#include <iostream>

template <typename T>
T add(T a, T b) {
    return a + b;
}

int main() {
    std::cout << "Int: " << add(2, 3) << std::endl;
    std::cout << "Doble: " << add(2.5, 3.5) << std::endl;
    return 0;
}

Templates de Clase


#include <iostream>

template <typename T>
class Box {
private:
    T value;

public:
    Box(T v) : value(v) {}

    T getValue() {
        return value;
    }
};

int main() {
    Box<int> intBox(123);
    Box<std::string> strBox("Hola");

    std::cout << "Valor entero: " << intBox.getValue() << std::endl;
    std::cout << "Valor de cadena: " << strBox.getValue() << std::endl;
    return 0;
}

Manejo de Archivos

Lectura desde un Archivo


#include <iostream>
#include <fstream>
#include <string>

int main() {
    std::ifstream infile("example.txt");
    std::string line;

    if (infile.is_open()) {
        while (std::getline(infile, line)) {
            std::cout << line << std::endl;
        }
        infile.close();
    } else {
        std::cout << "No se pudo abrir el archivo" << std::endl;
    }
    return 0;
}

Escritura en un Archivo


#include <iostream>
#include <fstream>

int main() {
    std::ofstream outfile("example.txt");

    if (outfile.is_open()) {
        outfile << "¡Hola, archivo!" << std::endl;
        outfile.close();
    } else {
        std::cout << "No se pudo abrir el archivo" << std::endl;
    }
    return 0;
}
