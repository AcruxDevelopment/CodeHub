Introducción

Java es un lenguaje de programación de alto nivel, basado en clases y orientado a objetos, que está diseñado para tener la menor cantidad de dependencias de implementación posibles. Es un lenguaje de programación de propósito general destinado a permitir a los desarrolladores de aplicaciones escribir una vez, ejecutar en cualquier lugar (WORA).

Sintaxis de Java

Estructura Básica


public class Main {
    public static void main(String[] args) {
        System.out.println("Hola, Mundo!");
    }
}

Conceptos de POO

Clases y Objetos


class Dog {
    String name;
    int age;

    Dog(String name, int age) {
        this.name = name;
        this.age = age;
    }

    void bark() {
        System.out.println("¡Guau!");
    }
}

public class Main {
    public static void main(String[] args) {
        Dog myDog = new Dog("Buddy", 5);
        myDog.bark();
        System.out.println(myDog.name + " tiene " + myDog.age + " años.");
    }
}

Herencia


class Animal {
    void eat() {
        System.out.println("Este animal come comida.");
    }
}

class Dog extends Animal {
    void bark() {
        System.out.println("¡Guau!");
    }
}

public class Main {
    public static void main(String[] args) {
        Dog myDog = new Dog();
        myDog.eat();
        myDog.bark();
    }
}

Polimorfismo


class Animal {
    void sound() {
        System.out.println("Este animal hace un sonido.");
    }
}

class Dog extends Animal {
    void sound() {
        System.out.println("¡Guau!");
    }
}

class Cat extends Animal {
    void sound() {
        System.out.println("¡Miau!");
    }
}

public class Main {
    public static void main(String[] args) {
        Animal myDog = new Dog();
        Animal myCat = new Cat();
        
        myDog.sound();
        myCat.sound();
    }
}

Encapsulación


class Person {
    private String name;
    private int age;

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }
}

public class Main {
    public static void main(String[] args) {
        Person person = new Person();
        person.setName("Juan");
        person.setAge(30);

        System.out.println(person.getName() + " tiene " + person.getAge() + " años.");
    }
}

Excepciones

Bloque Try-Catch


public class Main {
    public static void main(String[] args) {
        try {
            int division = 10 / 0;
        } catch (ArithmeticException e) {
            System.out.println("ArithmeticException: " + e.getMessage());
        }
    }
}

Lanzando Excepciones


class CustomException extends Exception {
    CustomException(String message) {
        super(message);
    }
}

public class Main {
    public static void main(String[] args) {
        try {
            throw new CustomException("Esta es una excepción personalizada");
        } catch (CustomException e) {
            System.out.println(e.getMessage());
        }
    }
}

Colecciones

Ejemplo de Lista


import java.util.ArrayList;
import java.util.List;

public class Main {
    public static void main(String[] args) {
        List list = new ArrayList<>();
        list.add("Manzana");
        list.add("Banana");
        list.add("Cereza");

        for (String fruit : list) {
            System.out.println(fruit);
        }
    }
}

Ejemplo de Mapa


import java.util.HashMap;
import java.util.Map;

public class Main {
    public static void main(String[] args) {
        Map map = new HashMap<>();
        map.put("Manzana", 1);
        map.put("Banana", 2);
        map.put("Cereza", 3);

        for (Map.Entry entry : map.entrySet()) {
            System.out.println(entry.getKey() + ": " + entry.getValue());
        }
    }
}

Streams

Ejemplo de Stream


import java.util.Arrays;
import java.util.List;

public class Main {
    public static void main(String[] args) {
        List numbers = Arrays.asList(1, 2, 3, 4, 5);

        int sum = numbers.stream()
                         .mapToInt(Integer::intValue)
                         .sum();

        System.out.println("Suma: " + sum);
    }
}

Filtrado y Mapeo


import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

public class Main {
    public static void main(String[] args) {
        List strings = Arrays.asList("abc", "", "bcd", "", "defg", "jk");

        List filtered = strings.stream()
                                       .filter(string -> !string.isEmpty())
                                       .collect(Collectors.toList());

        System.out.println(filtered);
    }
}

Concurrencia

Ejemplo de Thread


class MyThread extends Thread {
    public void run() {
        System.out.println("MyThread está corriendo");
    }

public class Main {
    public static void main(String[] args) {
        MyThread t = new MyThread();

Ejemplo de Runnable 

class MyRunnable implements Runnable {
    public void run() {
        System.out.println("MyRunnable está corriendo");
    }
}

public class Main {
    public static void main(String[] args) {
        Thread t = new Thread(new MyRunnable());
        t.start();
    }
}

Ejemplo de ExecutorService


import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;

public class Main {
    public static void main(String[] args) {
        ExecutorService executor = Executors.newFixedThreadPool(2);

        Runnable task1 = () -> {
            System.out.println("Task1 está corriendo");
        };

        Runnable task2 = () -> {
            System.out.println("Task2 está corriendo");
        };

        executor.submit(task1);
        executor.submit(task2);

        executor.shutdown();
    }
}
